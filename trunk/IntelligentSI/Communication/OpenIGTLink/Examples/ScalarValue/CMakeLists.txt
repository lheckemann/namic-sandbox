PROJECT(ScalarValue)

cmake_minimum_required(VERSION 2.4)
if(COMMAND cmake_policy)
  cmake_policy(SET CMP0003 NEW)
endif(COMMAND cmake_policy)

find_package(OpenIGTLink REQUIRED)

include(${OpenIGTLink_USE_FILE})

ADD_EXECUTABLE(ScalarValueClient   ScalarValueClient.cxx)
TARGET_LINK_LIBRARIES(ScalarValueClient OpenIGTLink)

ADD_EXECUTABLE(ScalarValueServer   ScalarValueServer.cxx)
TARGET_LINK_LIBRARIES(ScalarValueServer OpenIGTLink)
