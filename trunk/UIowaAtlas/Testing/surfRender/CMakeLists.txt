PROJECT (surfaceRenderVTKPolyData)

FIND_PACKAGE(VTK REQUIRED)
IF(NOT VTK_USE_RENDERING)
 MESSAGE(FATAL_ERROR "Example ${PROJECT_NAME} requires VTK_USE_RENDERING.")
ENDIF(NOT VTK_USE_RENDERING)
IF (VTK_FOUND)
       INCLUDE (${VTK_USE_FILE})
ENDIF (VTK_FOUND)


ADD_EXECUTABLE(surfaceRenderVTKPolyData surfaceRenderVTKPolyData.cxx)
TARGET_LINK_LIBRARIES (surfaceRenderVTKPolyData vtkRendering)
