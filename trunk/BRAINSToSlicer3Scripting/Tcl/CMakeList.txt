# Build a testing suite for Slicer-3
#
PROJECT(BRAINS_SLICER_TEST_TCL)


MESSAGE(STATUS "Configuring Slicer Tests")

SET(SLICER_COMMAND_INVOCATION "Slicer3 --no_splash -f")

SET(COMMON_INCLUDE_SCRIPT ${BRAINS_SLICER_TEST_SUITE_SOURCE_DIR}/Common/BRAINSWrappers.tcl )
SET(COMMON_DATA_DIR ${BRAINS_SLICER_TEST_SUITE_SOURCE_DIR}/Data )

SET(TESTNAME InvalidFileNameRead)
CONFIGURE_FILE(${BRAINS_SLICER_TEST_TCL_SOURCE_DIR}/${TESTNAME}.tcl.in ${EXECUTABLE_OUTPUT_PATH}/${TESTNAME}.tcl IMMEDIATE @ONLY )
ADD_TEST(${TESTNAME}_test  ${SLICER_COMMAND_INVOCATION}  ${EXECUTABLE_OUTPUT_PATH}/${TESTNAME}.tcl )
