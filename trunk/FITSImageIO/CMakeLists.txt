PROJECT(FITSImageIO)

# Find ITK.
FIND_PACKAGE(ITK REQUIRED)
IF(ITK_FOUND)
  INCLUDE(${ITK_USE_FILE})
ENDIF(ITK_FOUND)

SUBDIRS(
  cfitsio
  )

INCLUDE_DIRECTORIES(
  ${FITSImageIO_SOURCE_DIR}/cfitsio
  )

ADD_EXECUTABLE(FITSImageIOTest 
  FITSImageIOTest.cxx 
  itkFITSImageIO.cxx
  )

TARGET_LINK_LIBRARIES(FITSImageIOTest CFITSIO ITKIO ITKCommon)
